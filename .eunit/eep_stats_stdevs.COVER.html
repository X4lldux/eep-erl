        |  
        |  accumulate(State,X) ->
     4..|     N = State#eep_stats_stdevs.n + 1,
     4..|     D = X - State#eep_stats_stdevs.m,
     4..|     M = (D / N) + State#eep_stats_stdevs.m,
     4..|     M2 = State#eep_stats_stdevs.m2 + D*(X - M),
     4..|     State#eep_stats_stdevs{m2 = M2, m = M, d = D, n = N}.

     1..|     N = State#eep_stats_stdevs.n - 1,

     1..|     M = State#eep_stats_stdevs.m + (D / N),
     1..|     M2 = D*(X - M) + State#eep_stats_stdevs.m2,
     1..|     State#eep_stats_stdevs{m2 = M2, m = M, d = D, n = N}.
        |  
        |  emit(State) ->
     4..|    math:sqrt(State#eep_stats_stdevs.m2 / (State#eep_stats_stdevs.n - 1)).
